# Шаблон файла с переменными окружения
# Переименуйте этот файл в .env и заполните своими данными

# Refresh Token Alor (для WebSocket real-time данных)
ALOR_REFRESH_TOKEN=your_refresh_token_here

# API токен Alor (для HTTP запросов, устаревший)
ALOR_API_TOKEN=your_alor_token_here

# Настройки API Alor
ALOR_API_URL=https://api.alor.ru
ALOR_PORTFOLIO=your_portfolio_id

# Настройки cTrader FxPro Open API
CTRADER_CLIENT_ID=your_client_id_here
CTRADER_CLIENT_SECRET=your_client_secret_here
CTRADER_API_URL=https://api.spotware.com

# Дополнительные параметры cTrader (опционально)
CTRADER_ACCOUNT_ID=your_account_id_here
CTRADER_ACCESS_TOKEN=your_access_token_here
